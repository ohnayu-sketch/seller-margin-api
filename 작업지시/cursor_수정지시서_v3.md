# 수정 5 — 회계 탭: 간이 → 일반사업자 전환 경고

## 5-1. 연 매출 추적 + 전환 경고

```javascript
// 간이과세자 → 일반과세자 전환 기준
const THRESHOLDS = {
  warning: 64000000,   // 6,400만원: 주의 (연말 기준 8천만 초과 가능성)
  danger: 72000000,    // 7,200만원: 위험 (현재 추세로 초과 확실)
  limit: 80000000,     // 8,000만원: 일반과세자 전환 의무
};

function checkVatStatus(annualSales) {
  if (annualSales >= THRESHOLDS.limit) {
    return {
      status: 'danger',
      message: '⚠️ 연 매출 8,000만원 초과! 다음 해부터 일반과세자로 자동 전환됩니다.',
      action: '세무사 상담 필요'
    };
  } else if (annualSales >= THRESHOLDS.danger) {
    return {
      status: 'warning',
      message: '🔶 연 매출 8,000만원 초과 임박! 일반과세자 전환을 준비하세요.',
      action: '전환 시 세금계산서 발행 의무, 부가세 신고 방식 변경'
    };
  } else if (annualSales >= THRESHOLDS.warning) {
    return {
      status: 'notice',
      message: '📋 연 매출 6,400만원 돌파. 연말까지 추이를 확인하세요.',
      action: null
    };
  }
  return { status: 'safe', message: '✅ 간이과세자 유지 가능', action: null };
}
```

## 5-3. 일반과세자 전환이 오히려 유리한 경우 안내

매입 비중이 높을 때 일반과세자가 유리할 수 있음을 자동 계산해서 안내:

```javascript
function compareVatBenefit(annualSales, annualCost) {
  // 간이과세자 납부세액
  const simplifiedVat = annualSales * 0.015;
  
  // 일반과세자 납부세액 (매출세액 - 매입세액)
  const generalVat = (annualSales * 0.1) - (annualCost * 0.1);
  
  if (generalVat < simplifiedVat) {
    return {
      recommendation: '일반과세자가 유리',
      saving: simplifiedVat - generalVat,
      reason: `매입 비중이 높아 일반과세자 전환 시 연 ${formatNumber(simplifiedVat - generalVat)}원 절세 가능`
    };
  }
  return {
    recommendation: '간이과세자 유지',
    saving: generalVat - simplifiedVat,
    reason: `현재 구조에서 간이과세자가 연 ${formatNumber(generalVat - simplifiedVat)}원 유리`
  };
}
```
